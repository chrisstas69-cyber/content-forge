                return;
            }

            unprocessed.forEach(email => {
                const item = document.createElement('div');
                item.className = 'email-item';
                item.innerHTML = `
                    <div class="email-content">
                        <h4>${email.subject}</h4>
                        <p>From: ${email.from}</p>
                        <div class="link-preview">${email.link}</div>
                    </div>
                    <button class="btn btn-secondary" onclick="ingestVideo(${email.id})">⬇️ Download & Ingest</button>
                `;
                list.appendChild(item);
            });
        }

        function ingestVideo(emailId) {
            const email = emails.find(e => e.id === emailId);
            if(!email) return;

            showToast(`📥 Downloading video from ${email.from}...`);

            setTimeout(() => {
                email.processed = true;
                const newVideo = {
                    id: Date.now(),
                    name: `Ingested_${email.subject.replace(/\s+/g, '_')}.mp4`,
                    status: "01_Raw",
                    score: null,
                    caption: "",
                    tags: "",
                    date: new Date().toLocaleDateString(),
                    img: `https://picsum.photos/seed/${emailId}/300/500`
                };
                videos.unshift(newVideo); // Add to beginning of raw
                renderBoard();
                renderGmail(); // Remove from email list
                showToast("✅ Video added to 01_Raw folder!");
            }, 1000);
        }

        // --- Dashboard Logic ---
        function renderBoard() {
            // Clear lists
            ['01_Raw', '02_Ready_To_Post', '03_Needs_Review', '04_Published'].forEach(status => {
                document.getElementById(`list-${status.split('_')[1].toLowerCase()}`).innerHTML = '';
            });
            
            const counts = { '01_Raw': 0, '02_Ready_To_Post': 0, '03_Needs_Review': 0, '04_Published': 0 };

            videos.forEach(video => {
                counts[video.status]++;
                const listId = `list-${getShortStatus(video.status)}`;
                const card = document.createElement('div');
                card.className = `video-card status-${getShortStatus(video.status)}`;
                
                let scoreBadge = video.score !== null 
                    ? `<span class="viral-score ${getScoreClass(video.score)}">${video.score}/100</span>` 
                    : `<span class="viral-score" style="background:#9ca3af">--</span>`;

                card.innerHTML = `
                    <div class="card-header">
                        <span class="card-title">${video.name}</span>
                        ${scoreBadge}
                    </div>
                    <div class="card-meta">
                        <span>${video.date}</span>
                    </div>
                `;
                card.onclick = () => openVideoDetails(video.id);
                document.getElementById(listId).appendChild(card);
            });

            // Update Header Counts
            document.getElementById('count-raw').innerText = counts['01_Raw'];
            document.getElementById('count-ready').innerText = counts['02_Ready_To_Post'];
            document.getElementById('count-review').innerText = counts['03_Needs_Review'];
            document.getElementById('count-published').innerText = counts['04_Published'];

            updateStats();
        }

        function getShortStatus(status) {
            if(status === '01_Raw') return 'raw';
            if(status === '02_Ready_To_Post') return 'ready';
            if(status === '03_Needs_Review') return 'review';
            return 'published';
        }

        function getScoreClass(score) {
            return score >= 80 ? 'score-high' : (score >= 50 ? 'score-med' : 'score-low');
        }

        function updateStats() {
            document.getElementById('stat-total').innerText = videos.length;
            const scored = videos.filter(v => v.score !== null);
            const avg = scored.length ? (scored.reduce((a,b)=>a+b.score, 0) / scored.length).toFixed(0) : 0;
            document.getElementById('stat-score').innerText = avg;
            document.getElementById('stat-ready').innerText = videos.filter(v => v.status === '02_Ready_To_Post').length;
            document.getElementById('stat-review').innerText = videos.filter(v => v.status === '03_Needs_Review').length;
        }

        function runAutomationScan() {
            if(isProcessing) return;
            isProcessing = true;
            const btn = document.getElementById('run-scan-btn');
            btn.innerHTML = `<span>⏳</span> Processing...`;

            const rawVideos = videos.filter(v => v.status === '01_Raw');
            
            if(rawVideos.length === 0) {
                showToast("✅ No new videos in 01_Raw.");
                isProcessing = false;
                btn.innerHTML = `<span>▶</span> Run Automation Scan`;
                return;
            }

            showToast(`🔍 Analyzing ${rawVideos.length} videos...`);

            setTimeout(() => {
                rawVideos.forEach(video => {
                    video.score = Math.floor(Math.random() * (99 - 40 + 1) + 40);
                    video.caption = generateCaption(video.name);
                    video.tags = "#dogsoftiktok #viral #fyp";
                    
                    if (video.score >= 80) video.status = '02_Ready_To_Post';
                    else if (video.score >= 50) video.status = '03_Needs_Review';
                    // Else stays in raw
                });
                renderBoard();
                isProcessing = false;
                btn.innerHTML = `<span>▶</span> Run Automation Scan`;
                showToast("🚀 Scan Complete!");
            }, 1500);
        }

        // --- Modal & Editing Logic ---
        function openVideoDetails(id) {
            const video = videos.find(v => v.id === id);
            if(!video) return;

            currentVideoId = id;
            document.getElementById('modal-title').innerText = video.name;
            document.getElementById('modal-img').src = video.img;
            document.getElementById('modal-caption').value = video.caption || "";
            document.getElementById('modal-tags').value = video.tags || "";
            
            // Score
            document.getElementById('modal-score-text').innerText = video.score !== null ? video.score : "--";
            const scoreBar = document.getElementById('modal-score-bar');
            scoreBar.style.width = video.score ? `${video.score}%` : '0%';
            scoreBar.style.background = video.score >= 80 ? 'var(--success)' : (video.score >= 50 ? 'var(--warning)' : 'var(--danger)');

            // Apply Settings to Preview
            updatePreviewOverlay();

            // Actions
            const actions = document.getElementById('modal-actions');
            actions.innerHTML = '';

            if (video.status === '02_Ready_To_Post') {
                actions.innerHTML = `<button class="btn btn-primary" style="width:100%; justify-content:center;" onclick="publishVideo()">✅ Publish (Run Pipeline)</button>`;
            } else if (video.status === '03_Needs_Review') {
                actions.innerHTML = `<button class="btn btn-primary" style="width:100%; justify-content:center;" onclick="moveVideo('02_Ready_To_Post')">Approve</button>
                                    <button class="btn btn-secondary" style="width:100%; justify-content:center; margin-top:0.5rem; color:var(--danger);" onclick="moveVideo('01_Raw')">Reject</button>`;
            } else if (video.status === '04_Published') {
                actions.innerHTML = `<button class="btn btn-secondary" disabled>Published</button>`;
            } else {
                 actions.innerHTML = `<button class="btn btn-secondary" disabled>Run Scan first</button>`;
            }

            document.getElementById('video-modal').style.display = 'flex';
            // Reset Progress
            document.getElementById('modal-progress').style.display = 'none';
            document.getElementById('modal-progress-fill').style.width = '0%';
            document.getElementById('modal-progress-text').innerText = '';
        }

        function closeModal() {
            document.getElementById('video-modal').style.display = 'none';
        }

        // Update preview when settings change or user types
        document.getElementById('modal-caption').addEventListener('input', (e) => {
            const video = videos.find(v => v.id === currentVideoId);
            if(video) { video.caption = e.target.value; updatePreviewOverlay(); }
        });

        function updatePreviewOverlay() {
            const video = videos.find(v => v.id === currentVideoId);
            const wmEl = document.getElementById('preview-watermark');
            const capEl = document.getElementById('preview-caption');

            // Watermark
            if(settings.watermark) {
                wmEl.style.display = 'block';
                wmEl.innerText = settings.watermarkText;
            } else {
                wmEl.style.display = 'none';
            }

            // Caption
            if(settings.captions && video.caption) {
                capEl.style.display = 'block';
                capEl.innerText = video.caption;
            } else {
                capEl.style.display = 'none';
            }
        }

        function publishVideo() {
            const video = videos.find(v => v.id === currentVideoId);
            if(!video) return;

            const progressBar = document.getElementById('modal-progress');
            const progressFill = document.getElementById('modal-progress-fill');
            const progressText = document.getElementById('modal-progress-text');
            
            progressBar.style.display = 'block';
            
            // Define Pipeline Steps based on Settings
            const steps = [];
            if(settings.intro !== 'none') steps.push("Stitching Intro...");
            if(settings.trim) steps.push("Trimming Silence...");
            if(settings.captions) steps.push("Burning Captions...");
            if(settings.watermark) steps.push("Applying Watermark...");
            if(settings.music) steps.push("Mixing Audio...");
            if(settings.outro !== 'none') steps.push("Stitching Outro...");
            steps.push("Uploading to 04_Published...");
            steps.push("Logging to Sheets...");

            let stepIndex = 0;
            
            function runStep() {
                if(stepIndex >= steps.length) {
                    video.status = '04_Published';
                    renderBoard();
                    closeModal();
                    showToast("🎉 Video Published Successfully!");
                    return;
                }

                progressText.innerText = steps[stepIndex];
                progressFill.style.width = `${((stepIndex + 1) / steps.length) * 100}%`;
                
                stepIndex++;
                setTimeout(runStep, 600); // Simulate time per step
            }

            runStep();
        }

        function moveVideo(newStatus) {
            const video = videos.find(v => v.id === currentVideoId);
            if(video) {
                video.status = newStatus;
                renderBoard();
                closeModal();
                showToast("Moved to " + newStatus.replace('_', ' '));
            }
        }

        function generateCaption(name) {
            const templates = [
                "I can't believe he actually did that! 😂 #dogsoftiktok",
                "POV: When the treat bag opens. 🐕👂",
                "Wait for the zoomies at the end! 🏃💨",
                "Just a good boy being a good boy. ❤️",
                "Tag a friend who needs this today. 👇"
            ];
            return templates[Math.floor(Math.random() * templates.length)];
        }
